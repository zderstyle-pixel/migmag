function doGet(e) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName('MigMag'); if (!sheet) { sheet = ss.insertSheet('MigMag'); sheet.appendRow([ 'Date','Prénom','Nom','Téléphone','Code postal','Procédé', 'Situation','Employeur','CDI','Procédé principal', 'Matériaux','Positions','Certification','Lecture plans', 'Distance max','Horaires','Salaire actuel','Salaire visé', 'Disponibilité' ]); } var data = e.parameter; if (data.type === 'qualification') { var lastRow = sheet.getLastRow(); if (lastRow > 1) { sheet.getRange(lastRow, 10, 1, 10).setValues([[ data.procede_principal || '', data.materiaux || '', data.positions || '', data.certification || '', data.lecture_plans || '', data.distance_max || '', data.horaires || '', data.salaire_actuel || '', data.salaire_vise || '', data.disponibilite || '' ]]); } } else if (data.prenom) { sheet.appendRow([ new Date(), data.prenom || '', data.nom || '', data.tel || '', data.cp || '', data.procede || '', data.situation || '', data.employeur || '', data.cdi || '', '','','','','','','','','','' ]); } return ContentService .createTextOutput(JSON.stringify({status: 'ok'})) .setMimeType(ContentService.MimeType.JSON); }
